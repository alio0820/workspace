  function GetChangeDate(s_change_flag)
  {
    var txt_date=document.getElementById("txt_date");
    var s_date=txt_date.value;
    if (s_date == "")
    s_date=new Date();
    else
    s_date = new Date(s_date.replace("-","/"));
   
    if (s_change_flag=="Y")
    {
    s_date.setMonth(s_date.getMonth() + 1);
    txt_date.value=s_date.getYear()+"-"+(s_date.getMonth()+1)+"-"+s_date.getDate();
    }
