function   strlen(str)
dim   p_len
p_len=0
strlen=0
if   trim(str) <> " "   then
p_len=len(trim(str))
for   xx=1   to   p_len
if   asc(mid(str,xx,1)) <0   then
strlen=int(strlen)   +   2
else
strlen=int(strlen)   +   1
end   if
next
end   if
end   function

function   strvalue(str,lennum)

else
p_num=int(p_num)   +   1
end   if
strvalue=left(trim(str),x)& "… "
loop

end   function
